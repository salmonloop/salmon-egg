// 此文件应该在 iOS 平台项目中实现
// 文件路径: SalmonEgg.iOS/Storage/iOSSecureStorage.cs

using System;
using System.Text;
using System.Threading.Tasks;
using Foundation;
using Security;
using SalmonEgg.Infrastructure.Storage;

namespace SalmonEgg.iOS.Storage
{
    /// <summary>
    /// iOS 平台安全存储实现，使用 Keychain
    /// </summary>
    public class iOSSecureStorage : ISecureStorage
    {
        private const string ServiceName = "SalmonEgg";

        public async Task SaveAsync(string key, string value)
        {
            if (string.IsNullOrEmpty(key))
                throw new ArgumentNullException(nameof(key));
            if (value == null)
                throw new ArgumentNullException(nameof(value));

            await Task.Run(() =>
            {
                try
                {
                    // 删除旧记录（如果存在）
                    var existingRecord = new SecRecord(SecKind.GenericPassword)
                    {
                        Service = ServiceName,
                        Account = key
                    };
                    SecKeyChain.Remove(existingRecord);

                    // 添加新记录
                    var record = new SecRecord(SecKind.GenericPassword)
                    {
                        Service = ServiceName,
                        Account = key,
                        ValueData = NSData.FromString(value, NSStringEncoding.UTF8),
                        Accessible = SecAccessible.WhenUnlocked
                    };
                    
                    var result = SecKeyChain.Add(record);
                    if (result != SecStatusCode.Success)
                    {
                        throw new InvalidOperationException($"Failed to save to Keychain: {result}");
                    }
                }
                catch (Exception ex)
                {
                    throw new InvalidOperationException($"Failed to save secure data for key '{key}'", ex);
                }
            });
        }

        public async Task<string> LoadAsync(string key)
        {
            if (string.IsNullOrEmpty(key))
                throw new ArgumentNullException(nameof(key));

            return await Task.Run(() =>
            {
                try
                {
                    var record = new SecRecord(SecKind.GenericPassword)
                    {
                        Service = ServiceName,
                        Account = key
                    };
                    
                    var result = SecKeyChain.QueryAsRecord(record, out var resultCode);
                    
                    if (resultCode == SecStatusCode.ItemNotFound)
                        return null;
                    
                    if (resultCode != SecStatusCode.Success)
                        throw new InvalidOperationException($"Failed to load from Keychain: {resultCode}");
                    
                    return result.ValueData.ToString(NSStringEncoding.UTF8);
                }
                catch (Exception ex)
                {
                    throw new InvalidOperationException($"Failed to load secure data for key '{key}'", ex);
                }
            });
        }

        public async Task DeleteAsync(string key)
        {
            if (string.IsNullOrEmpty(key))
                throw new ArgumentNullException(nameof(key));

            await Task.Run(() =>
            {
                try
                {
                    var record = new SecRecord(SecKind.GenericPassword)
                    {
                        Service = ServiceName,
                        Account = key
                    };
                    
                    var result = SecKeyChain.Remove(record);
                    if (result != SecStatusCode.Success && result != SecStatusCode.ItemNotFound)
                    {
                        throw new InvalidOperationException($"Failed to delete from Keychain: {result}");
                    }
                }
                catch (Exception ex)
                {
                    throw new InvalidOperationException($"Failed to delete secure data for key '{key}'", ex);
                }
            });
        }
    }
}
